─── TAB SWITCHING JS (copy exactly) ───  
  
var TAB_BAR_TOP = 0;  
window.addEventListener('load', function () {  
 var tabBar = document.getElementById('tab-bar');  
 if (tabBar) {  
 var el = tabBar, top = 0;  
 while (el) { top += el.offsetTop; el = el.offsetParent; }  
 TAB_BAR_TOP = top;  
 }  
});  
function switchTab(tabId) {  
 document.querySelectorAll('.tab-content').forEach(function(el){ el.style.display = 'none'; });  
 document.querySelectorAll('.tab-btn').forEach(function(btn){ btn.classList.remove('active'); });  
 document.getElementById(tabId).style.display = 'block';  
 var activeBtn = document.querySelector('[data-tab="' + tabId + '"]');  
 if (activeBtn) { activeBtn.classList.add('active'); }  
 window.scrollTo({ top: TAB_BAR_TOP, behavior: 'smooth' });  
}  
  
Rules: TAB_BAR_TOP at outer scope · window 'load' event (not DOMContentLoaded) · id="tab-bar" on wrapper · class="tab-btn" + data-tab + onclick on buttons · class="tab-content" + id on panels
